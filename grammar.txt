stmt-list: (EOL | EOF) * (stmt (EOL | EOF)) *

stmt: if (expr) stmt else stmt
      if (expr) stmt
      { stmt-list }
     : expr

expr :  (KEYWORD: var) IDENTIFIER EQUAL expr
     :  IDENTIFIER EQUAL exprs
     :  comp-expr (TT_AND | TT_OR) comp-expr

comp-expr : TT_NOT comp-expr
          : arith-expr ((EE | NE | LT | GT | LTE | GTE) arith-expr) * 

arith-expr : term ((PLUS | MINUS) term) *

term : factor ((MUL | DIV) factor) *

factor: atom (^ atom) *

atom: (PLUS | MINUS) atom 
        :   INT | FLOAT | IDENTIFIER
        : LPAREN expr RPAREN
